```c++
#include <iostream>

using namespace std;
#define PI 3.14

class Cylinder {
private:
    double radius;
    double height;  

public:

    Cylinder(double r = 1.0, double h = 10.0) {
        radius = r;
        height = h;
    }


    double volume() {
        return PI * radius * radius * height;
    }

    void display() {
        cout << "圆柱体信息：半径：" << radius << " 高度：" << height << " 体积：" << volume() << endl;
    }
};

int main() {
    Cylinder c1; 
    c1.display();

    cout << endl;

    Cylinder c2(3.0, 5.0);  
    c2.display();

    return 0;
}
```

```c++
#include <iostream>
#include <cmath>
using namespace std;

class Line {
private:
    double x1, y1; 
    double x2, y2; 

public:
 
    Line(double a1, double b1, double a2, double b2) {
        x1 = a1;
        y1 = b1;
        x2 = a2;
        y2 = b2;
    }

 
    double distance() {
        return sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
    }

   
    void out() {
        cout << "线段端点坐标: (" << x1 << ", " << y1 << "), (" << x2 << ", " << y2 << ")" << endl;
    }
};

int main() {
    Line a(1, 0, 2, 0); 
    float disc;
    disc = a.distance(); 
    cout << "线段长度为：" << disc << endl;
    a.out(); 

    return 0;
}
```
