#include <iostream>

using namespace std;
#define PI 3.14

class Cylinder {
private:
    double radius;
    double height;  

public:

    Cylinder(double r = 1.0, double h = 10.0) {
        radius = r;
        height = h;
    }


    double volume() {
        return PI * radius * radius * height;
    }

    void display() {
        cout << "圆柱体信息：半径：" << radius << " 高度：" << height << " 体积：" << volume() << endl;
    }
};

int main() {
    Cylinder c1; 
    c1.display();

    cout << endl;

    Cylinder c2(3.0, 5.0);  
    c2.display();

    return 0;
}

#include <iostream>
#include <cmath>
using namespace std;

class Line {
private:
    double x1, y1; 
    double x2, y2; 

public:
 
    Line(double a1, double b1, double a2, double b2) {
        x1 = a1;
        y1 = b1;
        x2 = a2;
        y2 = b2;
    }

 
    double distance() {
        return sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
    }

   
    void out() {
        cout << "线段端点坐标: (" << x1 << ", " << y1 << "), (" << x2 << ", " << y2 << ")" << endl;
    }
};

int main() {
    Line a(1, 0, 2, 0); 
    float disc;
    disc = a.distance(); 
    cout << "线段长度为：" << disc << endl;
    a.out(); 

    return 0;
}

#include <iostream>
using namespace std;

class A {
	int x;
protected: int y;
public:
	int z;
	A(int a, int b, int c) { x = a; y = b; z = c; }//基类初始化
	int Getx() { return x; } //返回x
	int Gety() { return y; } //返回y
	void ShowA() { cout << "x=" << x << '\t' << " y=" << y << '\t' << "z=" << z << '\n'; }
};
class B :public A {

	int m, n;

public: B(int a, int b, int c, int d, int e) :A(a, b, c) { m = d; n = e; }

	  void Show() {
		  cout << "m = " << m << '\t' << "n = " << n << '\n';
		  cout << "x=" << Getx() << '\t' << "y =" << y << '\t' << "z=" << z << '\n';
	  }

	  int Sum() { return Getx() + y + z + m + n; }
};

int main(){
		B b1(1, 2, 3, 4, 5);

		b1.ShowA();
		b1.Show();
		cout << "Sum=="<< b1.Sum() << '\n';
		cout << "x=" << b1.Getx() << '\t';
		cout << "y=" << b1.Gety() << '\t';
		cout << "z="<< b1.z << '\n'; 
}
